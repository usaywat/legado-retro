# 存储管理界面设计文档

## 一、需求分析

### 1.1 功能需求

**核心功能**：显示可清理的缓存类型及其大小，支持一键清理和单独清理

**缓存类型**：

| 类型 | 说明 | 是否可展开 |
|------|------|------------|
| 书籍内容缓存 | 章节文本、漫画图片等阅读内容 | ✅ 是 |
| Epub 解压缓存 | Epub 格式书籍的解压临时文件 | ❌ 否 |
| 临时文件缓存 | 下载临时文件、解压临时目录等 | ❌ 否 |
| TTS 语音缓存 | 在线朗读引擎下载的语音文件 | ✅ 是 |
| ACache 磁盘缓存 | 书源变量、用户信息等运行时缓存 | ✅ 是 |
| 数据库缓存 | CacheDao 存储的临时数据记录 | ✅ 是 |
| 日志文件 | 应用运行日志、错误日志等 | ❌ 否 |

### 1.2 展开功能详情

**书籍内容缓存展开后显示**：
- 每本书的缓存大小
- 已缓存章节数
- 最后阅读时间
- 单独清理按钮

**TTS 语音缓存展开后显示**：
- 每个TTS引擎的缓存大小
- 最后使用时间
- 单独清理按钮

**ACache 磁盘缓存展开后显示**：
- 书源变量缓存（v_ 前缀）
- 用户信息缓存（userInfo_ 前缀）
- 登录Header缓存（loginHeader_ 前缀）
- 每项的记录数和大小
- 单独清理按钮

**数据库缓存展开后显示**：
- 各数据表的缓存大小
- 记录数
- 单独清理按钮

### 6.1 StorageCalculator 设计

```kotlin
object StorageCalculator {
    
    // ==================== 书籍缓存 ====================
    
    suspend fun calculateBookCacheSize(): Long = withContext(Dispatchers.IO) {
        val cacheDir = appCtx.externalFiles.getFile("book_cache")
        calculateDirSize(cacheDir)
    }
    
    suspend fun calculateBookCacheDetails(): List<CacheDetail> = withContext(Dispatchers.IO) {
        val cacheDir = appCtx.externalFiles.getFile("book_cache")
        val details = mutableListOf<CacheDetail>()
        
        // 遍历每本书的缓存文件夹
        cacheDir.listFiles()?.forEach { bookDir ->
            val book = findBookByFolderName(bookDir.name)
            if (book != null) {
                val size = calculateDirSize(bookDir)
                val chapterCount = countCachedChapters(book.bookUrl)
                details.add(CacheDetail(
                    id = book.bookUrl,
                    name = book.name,
                    meta = "${chapterCount}章 · 最后阅读: ${formatLastRead(book)}",
                    size = size,
                    formattedSize = ConvertUtils.formatFileSize(size)
                ))
            }
        }
        
        details.sortedByDescending { it.size }
    }
    
    fun clearBookCache(bookUrl: String? = null) {
        if (bookUrl == null) {
            BookHelp.clearCache()
        } else {
            val book = appDb.bookDao.getBook(bookUrl) ?: return
            BookHelp.clearCache(book)
        }
    }
    
    // ==================== TTS缓存 ====================
    
    suspend fun calculateTtsCacheSize(): Long = withContext(Dispatchers.IO) {
        val ttsDir = appCtx.cacheDir.getFile("httpTTS")
        calculateDirSize(ttsDir)
    }
    
    suspend fun calculateTtsCacheDetails(): List<CacheDetail> = withContext(Dispatchers.IO) {
        val ttsDir = appCtx.cacheDir.getFile("httpTTS")
        val details = mutableListOf<CacheDetail>()
        
        // 按TTS引擎分类
        ttsDir.listFiles()?.forEach { engineDir ->
            val size = calculateDirSize(engineDir)
            details.add(CacheDetail(
                id = engineDir.name,
                name = getTtsEngineName(engineDir.name),
                meta = "最后使用: ${formatLastUsed(engineDir)}",
                size = size,
                formattedSize = ConvertUtils.formatFileSize(size),
                iconColor = Color(0xFFEC4899)
            ))
        }
        
        details.sortedByDescending { it.size }
    }
    
    fun clearTtsCache(engineId: String? = null) {
        val ttsDir = appCtx.cacheDir.getFile("httpTTS")
        if (engineId == null) {
            FileUtils.delete(ttsDir.absolutePath)
        } else {
            FileUtils.delete(ttsDir.getFile(engineId).absolutePath)
        }
    }
    
    // ==================== ACache磁盘缓存 ====================
    
    suspend fun calculateACacheSize(): Long = withContext(Dispatchers.IO) {
        val aCacheDir = File(appCtx.cacheDir, "ACache")
        calculateDirSize(aCacheDir)
    }
    
    suspend fun calculateACacheDetails(): List<CacheDetail> = withContext(Dispatchers.IO) {
        val details = mutableListOf<CacheDetail>()
        
        // 按前缀分类统计
        val prefixes = listOf("v_", "userInfo_", "loginHeader_", "sourceVariable_")
        prefixes.forEach { prefix ->
            val (size, count) = calculateACacheByPrefix(prefix)
            if (count > 0) {
                details.add(CacheDetail(
                    id = prefix,
                    name = getACacheNameByPrefix(prefix),
                    meta = "${prefix}前缀 · ${count}条",
                    size = size,
                    formattedSize = ConvertUtils.formatFileSize(size),
                    iconColor = Color(0xFF10B981)
                ))
            }
        }
        
        details.sortedByDescending { it.size }
    }
    
    fun clearACache(prefix: String? = null) {
        if (prefix == null) {
            ACache.get().clear()
        } else {
            // 清理特定前缀的缓存
            CacheManager.clearByPrefix(prefix)
        }
    }
    
    // ==================== 数据库缓存 ====================
    
    suspend fun calculateDbCacheSize(): Long = withContext(Dispatchers.IO) {
        // 估算CacheDao的数据大小
        val count = appDb.cacheDao.count()
        count * 1024L  // 平均每条记录约1KB
    }
    
    suspend fun calculateDbCacheDetails(): List<CacheDetail> = withContext(Dispatchers.IO) {
        // 按key前缀分组统计
        val details = mutableListOf<CacheDetail>()
        
        // 这里需要根据实际的CacheDao实现来分组
        // 示例：按key前缀分组
        val groups = appDb.cacheDao.groupByPrefix()
        groups.forEach { group ->
            details.add(CacheDetail(
                id = group.prefix,
                name = getDbCacheName(group.prefix),
                meta = "${group.prefix}表 · ${group.count}条",
                size = group.size,
                formattedSize = ConvertUtils.formatFileSize(group.size),
                iconColor = Color(0xFF6366F1)
            ))
        }
        
        details.sortedByDescending { it.size }
    }
    
    suspend fun clearDbCache(prefix: String? = null) = withContext(Dispatchers.IO) {
        if (prefix == null) {
            appDb.cacheDao.clear()
        } else {
            appDb.cacheDao.deleteByPrefix(prefix)
        }
    }
    
    // ==================== 其他缓存 ====================
    
    suspend fun calculateEpubCacheSize(): Long = withContext(Dispatchers.IO) {
        val epubDir = appCtx.externalFiles.getFile("epub")
        calculateDirSize(epubDir)
    }
    
    fun clearEpubCache() {
        FileUtils.delete(appCtx.externalFiles.getFile("epub").absolutePath)
    }
    
    suspend fun calculateTempCacheSize(): Long = withContext(Dispatchers.IO) {
        calculateDirSize(appCtx.externalCache)
    }
    
    fun clearTempCache() {
        appCtx.externalCache.listFiles()?.forEach { it.deleteRecursively() }
    }
    
    suspend fun calculateLogCacheSize(): Long = withContext(Dispatchers.IO) {
        val logDir = appCtx.externalCache.getFile("log")
        calculateDirSize(logDir)
    }
    
    fun clearLogCache() {
        FileUtils.delete(appCtx.externalCache.getFile("log").absolutePath)
    }
    
    // ==================== 工具方法 ====================
    
    private fun calculateDirSize(dir: File): Long {
        if (!dir.exists()) return 0L
        var size = 0L
        dir.walkTopDown().forEach { file ->
            if (file.isFile) {
                size += file.length()
            }
        }
        return size
    }
}
```

### 6.2 ViewModel 设计

```kotlin
class StorageManageViewModel : ViewModel() {
    
    private val _uiState = MutableStateFlow<UiState>(UiState.Loading)
    val uiState: StateFlow<UiState> = _uiState.asStateFlow()
    
    private val _cacheItems = MutableStateFlow<List<CacheItem>>(emptyList())
    val cacheItems: StateFlow<List<CacheItem>> = _cacheItems.asStateFlow()
    
    val totalSize: StateFlow<Long> = _cacheItems
        .map { items -> items.sumOf { it.size } }
        .stateIn(viewModelScope, SharingStarted.Lazily, 0L)
    
    init {
        loadCacheInfo()
    }
    
    fun loadCacheInfo() {
        viewModelScope.launch {
            _uiState.value = UiState.Loading
            try {
                val items = listOf(
                    calculateCacheItem(CacheType.BOOK_CACHE),
                    calculateCacheItem(CacheType.EPUB_CACHE),
                    calculateCacheItem(CacheType.TEMP_CACHE),
                    calculateCacheItem(CacheType.TTS_CACHE),
                    calculateCacheItem(CacheType.ACACHE_DISK),
                    calculateCacheItem(CacheType.DB_CACHE),
                    calculateCacheItem(CacheType.LOG_CACHE)
                )
                _cacheItems.value = items
                _uiState.value = UiState.Idle
            } catch (e: Exception) {
                _uiState.value = UiState.Error(e.message ?: "加载失败")
            }
        }
    }
    
    fun toggleExpand(cacheType: CacheType) {
        viewModelScope.launch {
            val currentItems = _cacheItems.value.toMutableList()
            val index = currentItems.indexOfFirst { it.id == cacheType.name }
            if (index == -1) return@launch
            
            val item = currentItems[index]
            if (item.isExpanded) {
                // 折叠
                currentItems[index] = item.copy(isExpanded = false)
            } else {
                // 展开，加载详情
                val details = loadCacheDetails(cacheType)
                currentItems[index] = item.copy(
                    isExpanded = true,
                    details = details
                )
            }
            _cacheItems.value = currentItems
        }
    }
    
    fun clearCache(cacheType: CacheType, detailId: String? = null) {
        viewModelScope.launch {
            val target = detailId ?: cacheType.name
            _uiState.value = UiState.Clearing(target)
            try {
                when (cacheType) {
                    CacheType.BOOK_CACHE -> StorageCalculator.clearBookCache(detailId)
                    CacheType.EPUB_CACHE -> StorageCalculator.clearEpubCache()
                    CacheType.TEMP_CACHE -> StorageCalculator.clearTempCache()
                    CacheType.TTS_CACHE -> StorageCalculator.clearTtsCache(detailId)
                    CacheType.ACACHE_DISK -> StorageCalculator.clearACache(detailId)
                    CacheType.DB_CACHE -> StorageCalculator.clearDbCache(detailId)
                    CacheType.LOG_CACHE -> StorageCalculator.clearLogCache()
                }
                loadCacheInfo()
            } catch (e: Exception) {
                _uiState.value = UiState.Error(e.message ?: "清理失败")
            }
        }
    }
    
    fun clearAllCache() {
        viewModelScope.launch {
            _uiState.value = UiState.Clearing("所有缓存")
            try {
                CacheType.values().forEach { clearCache(it) }
                loadCacheInfo()
            } catch (e: Exception) {
                _uiState.value = UiState.Error(e.message ?: "清理失败")
            }
        }
    }
    
    private suspend fun calculateCacheItem(type: CacheType): CacheItem {
        val size = when (type) {
            CacheType.BOOK_CACHE -> StorageCalculator.calculateBookCacheSize()
            CacheType.EPUB_CACHE -> StorageCalculator.calculateEpubCacheSize()
            CacheType.TEMP_CACHE -> StorageCalculator.calculateTempCacheSize()
            CacheType.TTS_CACHE -> StorageCalculator.calculateTtsCacheSize()
            CacheType.ACACHE_DISK -> StorageCalculator.calculateACacheSize()
            CacheType.DB_CACHE -> StorageCalculator.calculateDbCacheSize()
            CacheType.LOG_CACHE -> StorageCalculator.calculateLogCacheSize()
        }
        
        val (canExpand, expandBadge) = when (type) {
            CacheType.BOOK_CACHE -> {
                val count = countCachedBooks()
                Pair(true, "${count}本")
            }
            CacheType.TTS_CACHE -> {
                val count = countTtsEngines()
                Pair(true, "${count}个引擎")
            }
            CacheType.ACACHE_DISK -> {
                val count = countACacheItems()
                Pair(true, "${count}项")
            }
            CacheType.DB_CACHE -> {
                val count = countDbTables()
                Pair(true, "${count}表")
            }
            else -> Pair(false, null)
        }
        
        return CacheItem(
            id = type.name,
            name = getCacheName(type),
            description = getCacheDescription(type),
            size = size,
            formattedSize = ConvertUtils.formatFileSize(size),
            icon = getCacheIcon(type),
            canExpand = canExpand,
            expandBadge = expandBadge
        )
    }
    
    private suspend fun loadCacheDetails(type: CacheType): List<CacheDetail> {
        return when (type) {
            CacheType.BOOK_CACHE -> StorageCalculator.calculateBookCacheDetails()
            CacheType.TTS_CACHE -> StorageCalculator.calculateTtsCacheDetails()
            CacheType.ACACHE_DISK -> StorageCalculator.calculateACacheDetails()
            CacheType.DB_CACHE -> StorageCalculator.calculateDbCacheDetails()
            else -> emptyList()
        }
    }
}
```

```kotlin
@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun StorageManageScreen(
    viewModel: StorageManageViewModel = viewModel(),
    onBackClick: () -> Unit
) {
    val uiState by viewModel.uiState.collectAsState()
    val cacheItems by viewModel.cacheItems.collectAsState()
    val totalSize by viewModel.totalSize.collectAsState()
    
    var showClearDialog by remember { mutableStateOf<ClearTarget?>(null) }
    var showClearAllDialog by remember { mutableStateOf(false) }
    
    Scaffold(
        topBar = {
            TopAppBar(
                title = { Text("存储管理") },
                navigationIcon = {
                    IconButton(onClick = onBackClick) {
                        Icon(Icons.Default.ArrowBack, "返回")
                    }
                },
                actions = {
                    IconButton(onClick = { showClearAllDialog = true }) {
                        Icon(Icons.Default.DeleteSweep, "一键清理")
                    }
                }
            )
        }
    ) { paddingValues ->
        when (uiState) {
            is UiState.Loading -> {
                Box(Modifier.fillMaxSize(), contentAlignment = Alignment.Center) {
                    CircularProgressIndicator()
                }
            }
            is UiState.Error -> {
                ErrorView((uiState as UiState.Error).message) {
                    viewModel.loadCacheInfo()
                }
            }
            else -> {
                LazyColumn(
                    modifier = Modifier.padding(paddingValues),
                    contentPadding = PaddingValues(16.dp),
                    verticalArrangement = Arrangement.spacedBy(12.dp)
                ) {
                    item {
                        CacheSummaryCard(
                            totalSize = totalSize,
                            itemCount = cacheItems.size
                        )
                    }
                    
                    items(cacheItems) { item ->
                        CacheItemCard(
                            item = item,
                            onExpandClick = { 
                                viewModel.toggleExpand(CacheType.valueOf(item.id))
                            },
                            onClearClick = { 
                                showClearDialog = ClearTarget(item.id, null)
                            },
                            onDetailClearClick = { detailId ->
                                showClearDialog = ClearTarget(item.id, detailId)
                            }
                        )
                    }
                }
            }
        }
    }
    
    // 清理确认对话框
    showClearDialog?.let { target ->
        ClearConfirmDialog(
            target = target,
            onConfirm = {
                viewModel.clearCache(
                    CacheType.valueOf(target.cacheType),
                    target.detailId
                )
                showClearDialog = null
            },
            onDismiss = { showClearDialog = null }
        )
    }
    
    // 一键清理对话框
    if (showClearAllDialog) {
        ClearAllConfirmDialog(
            onConfirm = {
                viewModel.clearAllCache()
                showClearAllDialog = false
            },
            onDismiss = { showClearAllDialog = false }
        )
    }
}
```

## 九、风险与应对

| 风险 | 影响程度 | 发生概率 | 应对措施 |
|------|----------|----------|----------|
| 大文件遍历耗时 | 中 | 中 | 使用协程异步计算，显示加载状态 |
| 清理过程中文件被占用 | 低 | 低 | 捕获异常，提示用户重试 |
| 缓存大小计算不精确 | 低 | 中 | 标注为"估算"，定期刷新 |
| 清理后应用异常 | 高 | 低 | 仅清理可安全清理的缓存类型 |
| 展开详情数据量大 | 低 | 低 | 限制显示数量，添加"查看更多" |

## 📋 总结

本设计基于项目现有的 Compose UI 模式（参考 UrlRecordScreen）和缓存管理机制（BookHelp、CacheManager）设计，采用 **Activity + Screen + ViewModel** 三层架构，无需引入新依赖。

**核心亮点**：
1. **展开功能**：支持查看每本书、每个引擎、每个分类的缓存详情
2. **单独清理**：每个详情项都可单独清理，精细化控制
3. **视觉设计**：彩色图标、渐变汇总卡片、流畅动画
4. **深色模式**：完整支持深色主题

**预计总开发时间：4-5天**
