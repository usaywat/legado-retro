## 🎯 需求背景

### 当前问题

应用板块（APP分类）的日志记录能力较弱，只能记录简单的文本消息，缺乏结构化信息，导致调试困难、问题定位慢。

### 目标

扩展应用板块的日志记录能力，使其能够像网络板块一样记录丰富的结构化信息，包括：
- 操作类型（做了什么）
- 操作对象（对谁做的）
- 操作结果（成功还是失败）
- 操作耗时（花了多长时间）
- 其他上下文信息

## 🔌 API设计

### 1. AppLog.kt 新增方法

**文件位置**: `app/src/main/java/io/legado/app/constant/AppLog.kt`

**新增方法**:

```kotlin
/**
 * 记录操作日志（扩展版）
 * 
 * @param operation 操作类型（如：备份、恢复、导入）
 * @param target 操作对象（如：书源名、文件名）
 * @param result 操作结果（成功、失败）
 * @param duration 操作耗时（毫秒）
 * @param extra 额外信息（键值对）
 * @param throwable 异常对象（可选）
 * @param toast 是否显示Toast提示
 */
fun putOperation(
    operation: String,
    target: String? = null,
    result: String? = null,
    duration: Long? = null,
    extra: Map<String, String>? = null,
    throwable: Throwable? = null,
    toast: Boolean = false
) {
    if (toast) {
        appCtx.toastOnUi("$operation $result")
    }
    
    // 构建日志消息
    val message = buildString {
        append(operation)
        target?.let { append(" $it") }
        result?.let { append(" $it") }
        duration?.let { append(" ${it}ms") }
    }
    
    // 构建详细信息
    val detail = buildString {
        operation.let { appendLine("操作: $it") }
        target?.let { appendLine("对象: $it") }
        result?.let { appendLine("结果: $it") }
        duration?.let { appendLine("耗时: ${it}ms") }
        extra?.forEach { (key, value) ->
            appendLine("$key: $value")
        }
        throwable?.let { 
            appendLine("异常: ${it.message}")
            append(it.stackTraceToString())
        }
    }
    
    // 上报到调试事件中心
    GlobalScope.launch(Dispatchers.IO) {
        DebugEventCenter.emit(
            DebugEvent(
                level = if (throwable != null) DebugLevel.ERROR else DebugLevel.INFO,
                category = DebugCategory.APP,
                message = message,
                detail = detail,
                operation = operation,
                target = target,
                result = result,
                duration = duration,
                extra = extra,
                throwable = throwable
            )
        )
    }
}

/**
 * 记录操作日志（简化版，自动计算耗时）
 * 
 * @param operation 操作类型
 * @param block 操作代码块
 * @return 操作结果
 */
inline fun <T> putOperationWithResult(
    operation: String,
    target: String? = null,
    extra: Map<String, String>? = null,
    block: () -> T
): T {
    val startTime = System.currentTimeMillis()
    var result: String? = null
    var throwable: Throwable? = null
    
    return try {
        val returnValue = block()
        result = "成功"
        returnValue
    } catch (e: Exception) {
        result = "失败"
        throwable = e
        throw e
    } finally {
        val duration = System.currentTimeMillis() - startTime
        putOperation(
            operation = operation,
            target = target,
            result = result,
            duration = duration,
            extra = extra,
            throwable = throwable
        )
    }
}
```

### 1. DebugLogDetailDialog.kt 扩展

```kotlin
when (log.category) {
    DebugCategory.NETWORK -> {
        // 网络板块处理（已有）
    }
    
    DebugCategory.APP -> {
        // 新增：应用板块操作详情
        if (log.operation != null || log.target != null || log.result != null || 
            log.duration != null || !log.extra.isNullOrEmpty()) {
            
            DetailSection(title = "操作详情", searchQuery = searchQuery) {
                log.operation?.let { 
                    DetailRow("操作类型", it, searchQuery) 
                }
                log.target?.let { 
                    DetailRow("操作对象", it, searchQuery) 
                }
                log.result?.let { 
                    DetailRow("操作结果", it, searchQuery) 
                }
                log.duration?.let { 
                    DetailRow("耗时", "${it}ms", searchQuery) 
                }
                
                // 显示额外信息
                log.extra?.forEach { (key, value) ->
                    DetailRow(key, value, searchQuery)
                }
            }
        }
    }
    
    DebugCategory.SOURCE, DebugCategory.RULE -> {
        // 书源板块处理（已有）
    }
    
    // ... 其他分类
}
```

// 使用自动计时方法
val importResult = AppLog.putOperationWithResult(
    operation = "导入书源",
    target = "书源合集.json",
    extra = mapOf(
        "来源" to "https://example.com/sources.json"

    // 执行导入逻辑
    val count = importSourcesFromFile("书源合集.json")
    count

// 或者手动记录
val startTime = System.currentTimeMillis()
try {
    val count = importSourcesFromFile("书源合集.json")
    val duration = System.currentTimeMillis() - startTime
    
    AppLog.putOperation(
        operation = "导入书源",
        target = "书源合集.json",
        result = "成功",
        duration = duration,
        extra = mapOf(
            "导入数量" to "$count",
            "成功数量" to "${count - 1}",
            "失败数量" to "1"
        )
    )
} catch (e: Exception) {
    val duration = System.currentTimeMillis() - startTime
    AppLog.putOperation(
        operation = "导入书源",
        target = "书源合集.json",
        result = "失败",
        duration = duration,
        throwable = e
    )
}
